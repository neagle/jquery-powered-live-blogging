Two plugins, actually.

  • $.liveUpdateApi
  • $.liveUpdateUi

$.liveUpdateApi

$.fn.liveUpdateApi = function () {
  return this.each(function () {
    // Hit the Live Update API with an AJAX call

    $(this).trigger('update', {
      // Data from the API
    });
  });
});

$.liveUpdateUi

$.fn.liveUpdateUi = function () {
  return this.each(function () {
    var $this = $(this);

    $this.liveUpdateApi();
    $this.bind('update', function (data) {
      // Build some UI with the incoming data
    });
  });
});

The API Plugin

Method / Instantiation Pattern

var methods = {
  init: function (options) {},
  fetch: function () {},
  pause: function () {},
  resume: function () {}
};

return this.each(function () {
  if (methods[method]) {
    return methods[method].apply(this, Array.prototype.slice.call(arguments, 1));
  } else if (typeof method === 'object' || ! method) {
    return methods.init.apply(this, args);
  } else {
    $.error('Method ' + method + ' does not exist on jQuery.liveBlogApi.');
  }
});
